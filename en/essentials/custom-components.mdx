export function BillingFlowStepper({ title, steps = [], alert }) {
  return (
    <div style={{ border: '1px solid #e2e8f0', borderRadius: '12px', padding: '24px', backgroundColor: '#ffffff' }}>
      {title && (
        <div style={{ fontSize: '14px', fontWeight: '600', color: '#374151', marginBottom: '20px' }}>{title}</div>
      )}
      <div style={{ display: 'flex', alignItems: 'flex-start' }}>
        {steps.map((step, i) => (
          <div key={i} style={{ flex: 1, display: 'flex', flexDirection: 'column', alignItems: 'flex-start' }}>
            <div style={{ display: 'flex', alignItems: 'center', width: '100%' }}>
              <div style={{
                width: '14px',
                height: '14px',
                borderRadius: '50%',
                backgroundColor: i === 0 ? '#1e3a8a' : '#cbd5e1',
                flexShrink: 0,
                zIndex: 1,
              }} />
              {i < steps.length - 1 && (
                <div style={{ flex: 1, height: '1px', backgroundColor: '#cbd5e1' }} />
              )}
            </div>
            <div style={{
              marginTop: '10px',
              paddingRight: '8px',
              fontSize: '13px',
              color: '#374151',
              fontWeight: i === 0 ? '500' : '400',
              lineHeight: '1.4',
            }}>
              {step}
            </div>
          </div>
        ))}
      </div>
      {alert && (
        <div style={{
          marginTop: '20px',
          border: '1px solid #fca5a5',
          borderLeft: '4px solid #ef4444',
          borderRadius: '8px',
          padding: '16px 16px 16px 20px',
          backgroundColor: '#fff5f5',
          display: 'flex',
          gap: '12px',
          alignItems: 'flex-start',
        }}>
          <span style={{ fontSize: '18px', flexShrink: 0, marginTop: '1px' }}>🛡</span>
          <div>
            <div style={{ fontSize: '14px', fontWeight: '600', color: '#991b1b', marginBottom: '8px' }}>{alert.title}</div>
            {alert.lines.map((line, i) => (
              <div key={i} style={{ fontSize: '13px', color: '#7f1d1d', lineHeight: '1.6', marginBottom: i < alert.lines.length - 1 ? '6px' : 0 }}>
                {line}
              </div>
            ))}
          </div>
        </div>
      )}
    </div>
  )
}

Waffo Docs provides a set of reusable React components that go beyond Mintlify's built-ins.
Use these when you need to match specific design patterns from the product design system.

## When to use custom vs built-in components

| Situation | Use |
|---|---|
| Simple callout, note, or warning | `<Note>`, `<Warning>` (Mintlify built-in) |
| Sequential steps | `<Steps>` (Mintlify built-in) |
| Feature cards with branded icon style | `CapabilityCard` (custom) |
| Payment method breakdown with logos | `PaymentMethodPanel` (custom) |
| Side-by-side model/plan comparison | `BusinessModelCard` (custom) |
| Horizontal flow diagram with alert | `BillingFlowStepper` (custom) |

## Adding a new component

1. Create `snippets/components/<your-component>.jsx` — keep the component code here as the source of truth.
2. In your `.mdx` page, copy the component function definition to the top of the file (after the frontmatter), then use it inline.
3. Add JSDoc at the top of the `.jsx` file with a usage example.
4. Add an entry to the table at the top of this page.
5. If you have a design file, tell Claude Code: "build a custom component from this design" — it will follow the Waffo design tokens automatically.

<Note>
Mintlify does not support `import` from `.jsx` snippet files at runtime. Always copy the component definition to the top of the MDX page where you use it. The `.jsx` files in `snippets/components/` serve as the **source of truth** for the component code — copy from there.
</Note>
